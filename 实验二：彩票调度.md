### 2.4.实现settickets的系统调用

根据实验说明修改proc.h中的struct proc如下：

```c
// Per-process state
struct proc {
  uint sz;                     // Size of process memory (bytes)
  pde_t* pgdir;                // Page table
  char *kstack;                // Bottom of kernel stack for this process
  enum procstate state;        // Process state
  int pid;                     // Process ID
  struct proc *parent;         // Parent process
  struct trapframe *tf;        // Trap frame for current syscall
  struct context *context;     // swtch() here to run process
  void *chan;                  // If non-zero, sleeping on chan
  int killed;                  // If non-zero, have been killed
  struct file *ofile[NOFILE];  // Open files
  struct inode *cwd;           // Current directory
  char name[16];               // Process name (debugging)
  int tickets;                 // 跟踪每个进程持有的彩票数量 
  int ticks;                    // 当前运行的CPU时间片
};
```

接着在proc.h中添加申明：

```c
int settickets(int);
```

显然，这个函数的作用是给进程设置票数，但是我要从哪里知道我现在在哪个进程呢？

幸好，在proc.c文件找到一个myproc()函数：
```c
myproc(void) {
  struct cpu *c;
  struct proc *p;
  pushcli();
  c = mycpu();
  p = c->proc;
  popcli();
  return p;
}
```
使用myproc可以获得当前进程的指针，于是可以顺利更改票数。

然后在proc.c文件中写出其完整代码：
```c
int settickets(int n){
  //不能少于1
  if (n < 1){ 
    return -1;
  }
  //设置当前进程的彩票数
  myproc()->tickets = n;
  return 0;
}
```

然后是在sysproc.c中编写sys_settickets()函数。

实验说明提到可以调用argint()函数，我在syscall.c中找到了定义：
```c
int
argint(int n, int *ip)
{
  return fetchint((myproc()->tf->esp) + 4 + 4*n, ip);
}
```
然后发现还有一个fetchint()函数不知道啥意思，

继续寻找其定义，在syscall.c中寻得
```c
int
fetchint(uint addr, int *ip)
{
  struct proc *curproc = myproc();

  if(addr >= curproc->sz || addr+4 > curproc->sz)
    return -1;
  *ip = *(int*)(addr);
  return 0;
}
```
结合来看，argint()函数的作用就是让内核能够安全地读取用户程序传递给系统调用的整数。

成功则返回0，否则就返回-1。

最终写得sys_settickets()函数如下：
```c
int sys_settickets(){
  int n;
  //未传入参数或传入参数小于1则返回-1，表示错误
  if(argint(0, &n) < 0||n < 1){
    return -1;
  }
  myproc()->tickets = n;
  return 0;
}
```
### 2.5.实现getpinfo系统调用
同样的，先在proc.h中进行声明：
```c
int getpinfo(struct pstat*);
```

接着在proc.c中完成完整的函数定义：
```c
int getpinfo(struct pstat*ps){
  struct proc *p;
  //检测是否传入空指针
  if(ps == 0){
    return -1;
  }
  
  //加锁
  acquire(&ptable.lock);
  
  for(int i=0;i<NPROC;i++){
    
    p=&ptable.proc[i];

    //填入每个进程的信息
    if(p->state != UNUSED){
      ps->inuse[i] = 1; //进程可运行
      ps->tickets[i] = p->tickets; //彩票数
      ps->pid[i] = p->pid; //进程ID
      ps->ticks[i] = p->ticks; //已经运行的CPU时间片数量
    }
    else{
      ps->inuse[i] = 0; //进程不可运行
      ps->tickets[i] = 0;
      ps->pid[i] = 0;
      ps->ticks[i] = 0;
    }
  }
  //解锁
  release(&ptable.lock);
  
  return 0;

}
```
其中锁的作用是保护别的进程不能修改ptable，保证读取的一致性。

即：ticks与tickets正确对应，而不是不同时间的两个数据拼凑在一起。

接着还要去sysproc.c写sys_getpinfo()

首先我去syscall.c中找到argptr()函数的定义：
```c
int
argptr(int n, char **pp, int size)
{
  int i;
  struct proc *curproc = myproc();
 
  if(argint(n, &i) < 0)
    return -1;
  if(size < 0 || (uint)i >= curproc->sz || (uint)i+size > curproc->sz)
    return -1;
  *pp = (char*)i;
  return 0;
}
```
与argint()函数类似，它的作用是安全地读取指针。

根据argptr()函数的定义来完成sys_getpinfo()函数的定义：
```c
int sys_getpinfo(){
  struct pstat* ps;
  //未传入参数或传入参数无效则返回-1，表示错误
  if(argptr(0, (void*)&ps, sizeof(struct pstat)) < 0){
    return -1;
  }
  
  return getpinfo(ps);
}
```

### 2.6.在创建和初始化进程时添加彩票信息：

文档中提到，默认情况下，每个进程被分配5个彩票。

同时显然知道初始ticks应为0。

所以需要在allocproc()函数中插入以下代码：
```c
p->tickets = 5; // 默认初始为5张票
p->ticks = 0; // 初始的CPU时间片为0
```
完整的allocproc()代码如下：
```c
static struct proc*
allocproc(void)
{
  struct proc *p;
  char *sp;

  acquire(&ptable.lock);

  for(p = ptable.proc; p < &ptable.proc[NPROC]; p++)
    if(p->state == UNUSED)
      goto found;

  release(&ptable.lock);
  return 0;

found:
  p->state = EMBRYO;
  p->pid = nextpid++;

  p->tickets = 5; // 默认初始为5张票
  p->ticks = 0; // 初始的CPU时间片为0

  release(&ptable.lock);

  // Allocate kernel stack.
  if((p->kstack = kalloc()) == 0){
    p->state = UNUSED;
    return 0;
  }
  sp = p->kstack + KSTACKSIZE;

  // Leave room for trap frame.
  sp -= sizeof *p->tf;
  p->tf = (struct trapframe*)sp;

  // Set up new context to start executing at forkret,
  // which returns to trapret.
  sp -= 4;
  *(uint*)sp = (uint)trapret;

  sp -= sizeof *p->context;
  p->context = (struct context*)sp;
  memset(p->context, 0, sizeof *p->context);
  p->context->eip = (uint)forkret;

  return p;
}
```

而对于fork()函数，则需要“在创建进程时需要为其分配彩票数。具体来说，确保子进程在创建时继承父进程的彩票数量。”

因此需要加入以下代码：

```c
np->tickets = curproc->tickets;   // 继承父进程的彩票数
np->ticks = 0;                     // 子进程cpu时间片从 0 开始
```

完整的fork()函数代码如下：
```c
int
fork(void)
{
  int i, pid;
  struct proc *np;
  struct proc *curproc = myproc();

  // Allocate process.
  if((np = allocproc()) == 0){
    return -1;
  }

  // Copy process state from proc.
  if((np->pgdir = copyuvm(curproc->pgdir, curproc->sz)) == 0){
    kfree(np->kstack);
    np->kstack = 0;
    np->state = UNUSED;
    return -1;
  }
  np->sz = curproc->sz;
  np->parent = curproc;
  *np->tf = *curproc->tf;

  // Clear %eax so that fork returns 0 in the child.
  np->tf->eax = 0;

  for(i = 0; i < NOFILE; i++)
    if(curproc->ofile[i])
      np->ofile[i] = filedup(curproc->ofile[i]);
  np->cwd = idup(curproc->cwd);

  safestrcpy(np->name, curproc->name, sizeof(curproc->name));

  np->tickets = curproc->tickets;   // 继承父进程的彩票数
  np->ticks = 0;                     // 子进程时间片从 0 开始
  
  pid = np->pid;

  acquire(&ptable.lock);

  np->state = RUNNABLE;

  release(&ptable.lock);

  return pid;
}
```

### 2.7.调整调度算法

彩票调度算法可以理解为把所有进程的彩票数排成一列，然后随机抽取一个数作为中奖号码

持有该号码的进程获得cpu时间

根据该思路可以修改sheduler()函数如下：
```c
void
scheduler(void)
{
  struct proc *p;
  struct cpu *c = mycpu();
  c->proc = 0;
  
  for(;;){
    // Enable interrupts on this processor.
    sti();

    // Loop over process table looking for process to run.
    acquire(&ptable.lock);

    int totals=0;// 计算总票数

    for(p = ptable.proc; p < &ptable.proc[NPROC]; p++){
      if(p->state == RUNNABLE)
        totals += p->tickets;  // 累加所有可运行进程的票数
    }
    if(totals > 0){
      int winner = random_at_most(totals-1); //
      int count = 0;
      for(p = ptable.proc; p < &ptable.proc[NPROC]; p++){
        if(p->state == RUNNABLE){
          count += p->tickets; // 累加当前进程的票数 
          if(count > winner){ // 当累加的票数超过了随机数，说明中奖
            c->proc = p;
            switchuvm(p);
            p->state = RUNNING;

            p->ticks++; // 运行一次，CPU时间片加1
            
            swtch(&(c->scheduler), p->context);
            switchkvm();

            // Process is done running for now.
            // It should have changed its p->state before coming back.
            c->proc = 0;
            break;
          }
        }
      }
    }
    release(&ptable.lock);

  }
}
```
